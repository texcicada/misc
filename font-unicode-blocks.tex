	\fi

	\int_gincr:N 
		\g_uc_glyphtotal_int

	\iffontchar\font #1 
		\bool_gset_true:N 
			\g_uc_blockexists_bool

%=================
\cs_new_protected:Npn \func_printablock:n #1 {

%		#1 = (,)name;start..finish

%	get the block info  ========----------------
		\seq_gset_split:Nnn
			\g_uc_blockrecord_seq
			{ ; }
			{ #1 }
%	get the block name  ========----------------
		\tl_set:Nx
			\g_uc_blockname_tl
			{
				\seq_item:Nn
					\g_uc_blockrecord_seq
					{ 1 }

%%			\g_uc_blockname_tl

%	get the block range data  ========----------------
		\tl_set:Nx
			\g_uc_blockrange_tl
			{
				\seq_item:Nn
					\g_uc_blockrecord_seq
					{ 2 }
			}
%%		\tl_show:N
%%			\g_uc_blockrange_tl

%	get the block range info  ========----------------
		\seq_gset_split:NnV
			\g_uc_blockrange_seq
			{ .. }
			\g_uc_blockrange_tl 
%%		\seq_show:N
%%			\g_uc_blockrange_seq

%		get the block start  ========----------------
		\tl_set:Nx
			\g_uc_blockstart_tl
			{
				\seq_item:Nn
					\g_uc_blockrange_seq
					{ 1 }
			}

%%			\g_uc_blockstart_tl

			\g_uc_blockstart_int
			{
				\int_from_hex:o{
					\g_uc_blockstart_tl
					}
			}
%%		\int_show:N
%%			\g_uc_blockstart_int

%		get the block finish  ========----------------
		\tl_set:Nx
			\g_uc_blockfinish_tl
			{
				\seq_item:Nn
					\g_uc_blockrange_seq
					{ 2 }
			}
%%		\tl_show:N
%%			\g_uc_blockfinish_tl
		\int_set:Nn
			\g_uc_blockfinish_int
			{
				\int_from_hex:o{
					\g_uc_blockfinish_tl
					}
			}
%		\int_show:N
%			\g_uc_blockfinish_int
	\bool_gset_false:N 
		\g_uc_blockexists_bool
		
\int_gset:Nn 
		\g_uc_glyphcount_int
		{ 0 }
\int_gset:Nn 
		\g_uc_glyphtotal_int
		{ 0 }
		
	\int_step_function:nnN
		{ \g_uc_blockstart_int }
		{ \g_uc_blockfinish_int }
		\func_checkablock:n
	
	
	% blockname -----------------
	\bool_if:NT
	\g_uc_blockexists_bool
	{

		\group_begin:
		\tex_par:D
		\sffamily
		\color{blue}
		\noindent
		\tl_use:N
			\g_uc_blockname_tl
		\group_end:

	\begin{quotation}\noindent
	\marginpar{{
				\footnotesize\ttfamily
				\int_use:N 
					\g_uc_glyphcount_int
				\space of ~

					\g_uc_glyphtotal_int

		{ \g_uc_blockfinish_int }

	}
